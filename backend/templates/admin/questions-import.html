Key Type Description
ageGroups string[] Ordered list of age group labels
skinTypes map<age, string[]> Skin type options per age group
concerns map<age, string[]> Concern options per age group
productMapping map<age, map<concern, brand>> Brand recommendation per concern per age group
priority map<age, string[]> Brand priority order for product selection

Example structure

{ "ageGroups": ["Under 25", "25-35"], "skinTypes": { "Under 25": ["Dry", "Normal", "Oily/Combination"], "25-35": ["Dry", "Normal", "Oily/Combination", "Mature"] }, "concerns": { "Under 25": ["Dark spots/Uneven skin tone", "Oil control"], "25-35": ["Fine lines and wrinkles", "Loss of firmness"] }, "productMapping": { "Under 25": { "Dark spots/Uneven skin tone": "Glow", "Oil control": "Glow" }, "25-35": { "Fine lines and wrinkles": "Rise", "Loss of firmness": "Rise" } }, "priority": { "Under 25": ["Glow", "Silk"], "25-35": ["Rise", "Glow"] } }
